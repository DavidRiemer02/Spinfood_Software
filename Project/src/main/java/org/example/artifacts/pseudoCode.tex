    \section{PairMatchAlgorithm}
    \begin{algorithm}
        \caption{PairMatch}
        \label{euclid}
        \begin{algorithmic}[1]
            \Procedure{Algorithmus~um~Paare~zu~erstellen}{}
                \State $ \text{Erstelle~einen~ungerichteten~Graphen \textit{G$\{$V,E$\}$},~mit~Teilnehmnern~als~Knoten}$
                \For{\textit{alle~Kombinationen~an~Knoten~V~in~G}}
                    \State \text{überprüfe~ob}~\textit{fullfillsHardCriteria=true}
                    \If {\textit{fullfillsHardCriteria=true}}
                        \State \text{berechne~Kosten~der~jeweiligen~Kriterien}
                        \State \text{Füge~Kante~zwischen~den~Solo-Paaren~hinzu,~mit~den~Kosten~als~Gewicht}
                    \EndIf
                \EndFor
                \For{\textit{alle~Knoten~V~in~G}}
                    \If {\textit{noch~Solos~in~V~enthalten}}
                        \State \text{Solo~mit~geringster~Anzahl~an~Kanten~wird~gewählt}
                        \State\text{Finde~Kante~mit~niedrigstem~Gewicht~welche~von~diesem~Solo~ausgeht}
                        \State\text{Ein~Paar~Objekt~wird~aus~diesen~beiden~Solos~erstellt~und~in~eine~Liste~PairMatched~hinzugefügt}
                        \State\text{Entferne~beide~Solos~aus~$V$~und~deren~Kante~aus~$E$}
                    \EndIf
                \EndFor
                \State \Return \text{PairMatched}
            \EndProcedure
        \end{algorithmic}
    \end{algorithm}

    \section{GroupMatchAlgorithm}
    \begin{algorithm}
        \caption{GroupMatch}
        \label{euclid}
        \begin{algorithmic}[1]
            \Procedure{Algorithmus~um~Gruppen~zu~erstellen}{}
                \State $ \text{Unterteile~die~Paare~in~Meat/None~und~Vegan/Veggie~Listen}$
                \For{\textit{Beide~Listen}}
                    \State \text{Erstelle~2~Graphen~$G_1$~und~$G_2$,~}
                \EndFor
                \For{\textit{alle~Kombinationen~an~Knoten~$V_1$~in~$G_1$~und~$V_2$~in~$G_2$}}
                    \If {\textit{fullfillsHardCriteria=true}}
                        \State \text{berechne~Kosten~der~jeweiligen~Kriterien}
                        \State \text{Füge~Kante~zwischen~den~"Pärchen-Paaren"~hinzu,~mit~den~Kosten~als~Gewicht}
                    \EndIf
                \EndFor
                \For{\textit{alle~Knoten~$V_1$~in~$G_1$~und~$V_2$~in~$G_2$}}
                    \If {\textit{noch~Pärchen~mit~8~Kanten~entalten}}
                        \State \text{Pärchen~mit~geringster~Anzahl~an~Kanten~wird~gewählt~(mind.~8~Kanten)}
                        \State\text{Finde~8~Kanten~mit~niedrigstem~Gewicht~welche~von~diesem~Paar~ausgehen}
                        \State\text{Erstelle~$SuperGroup$~Objekt~mit~9~Pärchen}
                        \State\text{Aus~der~$SuperGroup$~werden~neun~ 3er~
                        Gruppen erstellt,~ wobei~ jedes ~Pärchen~ einmal~ kocht~ und ~bei~ jedem ~Gang ~sich~ unterschiedliche ~Pärchen ~treffen}
                        \State\text{Entferne~die~9~Pärchen~aus~den~jeweiligen~Graphen}
                    \EndIf
                \EndFor
                \State \Return \text{GroupMatched}
            \EndProcedure
        \end{algorithmic}
    \end{algorithm}
